    // ==========================================
    // ACTION FUNCTIONS (Via Centrale Router)
    // ==========================================

    function uiSaveAdminUser(userId) {
        const status = document.getElementById('ustatus-' + userId).value;
        const role = document.getElementById('urole-' + userId).value;
        const linkedId = document.getElementById('ulink-' + userId).value;

        google.script.run
            .withSuccessHandler(() => { showToast('Gebruiker geüpdatet!', 'success'); Splash.init(); })
            .withFailureHandler(err => showToast('Fout: ' + err.message, 'error'))
            .handlePluginAction('HQ', 'adminUpdateUser', { userId, status, role, linkedId });
    }

    function uiAddHQAnnouncement(event) {
        const title = document.getElementById('hq-ann-title').value;
        const content = document.getElementById('hq-ann-content').value;
        if(!title || !content) return showToast('Vul alle velden in!', 'error');
        
        const btn = event.target; btn.disabled = true; btn.innerText = '...';
        google.script.run
            .withSuccessHandler(() => { showToast('Announcement geplaatst', 'success'); Splash.init(); navigate('hq'); })
            .withFailureHandler(err => { showToast('Fout: ' + err.message, 'error'); btn.disabled = false; })
            .handlePluginAction('HQ', 'addAnnouncement', { leagueId: getLeague().id, title, content });
    }

    function uiDeleteAnnouncement(id) {
        google.script.run
            .withSuccessHandler(() => { showToast('Verwijderd!', 'info'); Splash.init(); navigate('hq'); })
            .withFailureHandler(err => showToast('Fout: ' + err.message, 'error'))
            .handlePluginAction('HQ', 'deleteAnnouncement', { id });
    }

    function uiAddNote(event) {
        const content = document.getElementById('hq-note-content').value;
        if(!content) return showToast('Notitie is leeg!', 'error');
        
        const btn = event.target; btn.disabled = true;
        google.script.run
            .withSuccessHandler(() => { showToast('Notitie opgeslagen', 'success'); Splash.init(); navigate('hq'); })
            .withFailureHandler(err => { showToast('Fout: ' + err.message, 'error'); btn.disabled = false; })
            .handlePluginAction('HQ', 'addNote', { leagueId: getLeague().id, authorId: AppState.currentUser.id, content });
    }

    function uiDeleteNote(id) {
        google.script.run
            .withSuccessHandler(() => { Splash.init(); navigate('hq'); })
            .withFailureHandler(err => showToast('Fout: ' + err.message, 'error'))
            .handlePluginAction('HQ', 'deleteNote', { id });
    }
