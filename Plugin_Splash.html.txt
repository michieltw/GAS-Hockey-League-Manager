<script>
  const Splash = {
    init: function() {
      // Haal nu in één keer álle data op voor de hele app via je Core Router!
      google.script.run
        .withSuccessHandler(this.onDataLoaded.bind(this))
        .withFailureHandler(function(error) {
                document.getElementById('splash-loading').innerHTML = `<p class="text-red-500 font-bold">Fout bij laden:<br>${error.message}</p>`;
            })
            .handlePluginAction('Core', 'getInitialData', null); // <--- GOED
    },

    onDataLoaded: function(serverData) {
      if (serverData && serverData.error) {
        this.onError(serverData.error);
        return;
      }

      // 1. Sla de data globaal op in de 'Motor' (Plugin_Core.html)
      MockData = serverData;
      
      // 2. Bouw de League Switcher in je zijbalk op (als je competities hebt)
      const switcher = document.getElementById('league-switcher');
      if (switcher) {
          let optionsHtml = `<option value="all">Alle Competities</option>`;
          if(MockData.leagues && MockData.leagues.length > 0) {
              optionsHtml += MockData.leagues.map(l => `<option value="${l.id}">${l.name}</option>`).join('');
          }
          switcher.innerHTML = optionsHtml;
          switcher.value = AppState.currentLeagueId;
      }

      // Voorbeeld van waar je het aanroept:
      function onDataLoaded(data) {
      MockData = data;
    
    // Start de preloader
    renderPreloader(document.getElementById('page_preloader'));
    
    // Verder met renderen...
}

      // Zorg dat de sidebar gebruiker up-to-date is
      if(typeof updateSidebarUser === 'function') updateSidebarUser();

      document.getElementById('splash-status').innerText = "Data geladen. Systeem gereed.";
      
      // 3. Ga na 1 seconde naar het Dashboard en teken de data!
      setTimeout(() => {
        App.showApp(); // Verbergt splash, toont wrapper
        navigate('dashboard'); // Jouw nieuwe router functie die het dashboard tekent
      }, 1000);
    },

    onError: function(err) {
      document.getElementById('splash-status').innerHTML = `<span class="text-red-500 font-bold">Fout: ${err}</span>`;
    }
  };

  // Start direct zodra de plugin geladen is
  Splash.init();
</script>
