<!-- De Container waar je router naar zoekt -->
<div id="page_dashboard" class="plugin-page active h-full"></div>

<script>
    // Jouw letterlijke renderDashboard functie
    function renderDashboard(container) {
        const league = getLeague();
        const teams = getLeagueTeams();
        const players = MockData.players || [];
        const schedule = AppState.currentLeagueId === 'all' ? (MockData.schedule || []) : (MockData.schedule || []).filter(s => s.leagueId === AppState.currentLeagueId);
        const announcements = AppState.currentLeagueId === 'all' ? (MockData.announcements || []) : (MockData.announcements || []).filter(a => a.leagueId === AppState.currentLeagueId);

        const bannerStyle = (league.bannerUrl && league.bannerUrl.trim() !== '') ? `background-image: url('${league.bannerUrl}'); background-size: cover; background-position: center;` : 'background-color: hsl(0 0% 12%);';
        const logoHtml = (league.logoUrl && league.logoUrl.trim() !== '')
            ? `<img src="${league.logoUrl}" class="w-32 h-32 md:w-40 md:h-40 shrink-0 object-contain drop-shadow-2xl" alt="League Logo" />`
            : `<div class="w-28 h-28 md:w-40 md:h-40 rounded-[2rem] shadow-2xl bg-secondary flex items-center justify-center font-oswald text-4xl border-2 border-border shrink-0 text-white">${league.name.substring(0,3).toUpperCase()}</div>`;

        let announcementsHtml = '';
        if (announcements.length > 0) {
            const sortedAnnouncements = announcements.sort((a, b) => new Date(b.date) - new Date(a.date));
            announcementsHtml = sortedAnnouncements.map(a => {
                const annLeague = MockData.leagues.find(l => l.id === a.leagueId);
                const badge = AppState.currentLeagueId === 'all' && annLeague ? `<span class="bg-white/10 text-white px-2 py-0.5 rounded text-[8px] ml-2 inline-block">${annLeague.name}</span>` : '';
                return `
                <div class="border-b border-border/50 last:border-0 pb-4 last:pb-0 mb-4 last:mb-0 hover:bg-white/[0.02] p-2 -mx-2 rounded-lg transition-colors">
                    <p class="text-[10px] text-papaya font-bold uppercase tracking-widest mb-1">${a.date} ${badge}</p>
                    <h3 class="font-bold text-white text-lg mb-1">${a.title}</h3>
                    <p class="text-sm text-muted leading-relaxed">${a.content}</p>
                </div>`}).join('');
        } else {
            announcementsHtml = `<p class="text-sm text-muted italic">Momenteel geen mededelingen.</p>`;
        }

        let tasksHtml = '';
        let tasksFormHtml = '';
        const isManagerOrOfficer = AppState.currentUser && (AppState.currentUser.role === 'Manager' || AppState.currentUser.role === 'Officer');
        const tasksList = (MockData.tasks || []).filter(t => {
            if(AppState.currentLeagueId !== 'all' && t.leagueId !== AppState.currentLeagueId) return false;
            if(isManagerOrOfficer) return true;
            if(AppState.currentUser && t.assignedTo === AppState.currentUser.id) return true;
            return false;
        });

        if (AppState.currentUser) {
            if (isManagerOrOfficer) {
                const usersOptions = MockData.users.map(u => `<option value="${u.id}">${u.name || u.username}</option>`).join('');
                tasksFormHtml = `
                <div class="bg-secondary/50 border border-border p-4 rounded-xl mb-4">
                    <h3 class="text-[10px] font-bold text-white mb-2 uppercase tracking-widest">Nieuwe Taak Toewijzen</h3>
                    <div class="grid grid-cols-1 md:grid-cols-2 gap-2 mb-2">
                        <input type="text" id="new-task-title" placeholder="Titel" class="bg-secondary border border-border rounded-lg p-2 text-xs text-white outline-none">
                        <select id="new-task-assignee" class="bg-secondary border border-border rounded-lg p-2 text-xs text-white outline-none">
                            <option value="">Toewijzen aan...</option>
                            ${usersOptions}
                        </select>
                        <input type="date" id="new-task-deadline" class="bg-secondary border border-border rounded-lg p-2 text-xs text-white outline-none">
                        <input type="text" id="new-task-desc" placeholder="Beschrijving (optioneel)" class="bg-secondary border border-border rounded-lg p-2 text-xs text-white outline-none">
                    </div>
                    <button onclick="uiAddTask(event)" class="w-full py-2 bg-papaya/20 text-papaya border border-papaya/50 font-bold uppercase tracking-widest rounded-lg hover:bg-papaya hover:text-black transition-colors text-[10px]">Taak Aanmaken</button>
                </div>`;
            }

            if (tasksList.length > 0) {
                tasksHtml = tasksList.map(t => {
                    const assignedUser = MockData.users.find(u => u.id === t.assignedTo);
                    const assignedName = assignedUser ? (assignedUser.name || assignedUser.username) : 'Onbekend';
                    const isDone = t.status === 'Voltooid';
                    const canEdit = isManagerOrOfficer || (AppState.currentUser && t.assignedTo === AppState.currentUser.id);
                    const editBtn = isManagerOrOfficer ? `<button onclick="uiEditTask('${t.id}')" class="w-6 h-6 rounded bg-blue-500/20 text-blue-400 flex items-center justify-center hover:bg-blue-500 hover:text-white transition-colors" title="Bewerk Taak"><i data-lucide="edit-3" class="w-3 h-3"></i></button>` : '';
                    const deleteBtn = isManagerOrOfficer ? `<button onclick="uiDeleteTask('${t.id}')" class="w-6 h-6 rounded bg-red-500/20 text-red-400 flex items-center justify-center hover:bg-red-500 hover:text-white transition-colors" title="Verwijder Taak"><i data-lucide="trash-2" class="w-3 h-3"></i></button>` : '';

                    return `
                    <div class="flex items-start gap-3 p-3 rounded-xl border border-border bg-card shadow-sm hover:border-papaya/30 transition-colors ${isDone ? 'opacity-60' : ''}">
                        ${canEdit ? `<input type="checkbox" onchange="uiToggleTaskStatus('${t.id}', '${t.status}')" class="mt-1 w-4 h-4 rounded bg-secondary border-border focus:ring-papaya text-papaya cursor-pointer" ${isDone ? 'checked' : ''}>` : `<div class="mt-1 w-4 h-4 rounded border border-border flex items-center justify-center">${isDone ? '<i data-lucide="check" class="w-3 h-3 text-papaya"></i>' : ''}</div>`}
                        <div class="flex-1 min-w-0">
                            <p class="text-sm font-bold text-white ${isDone ? 'line-through text-muted' : ''}">${t.title}</p>
                            ${t.description ? `<p class="text-xs text-muted mt-1 leading-tight">${t.description}</p>` : ''}
                            <div class="flex flex-wrap items-center gap-2 mt-2">
                                <span class="text-[9px] uppercase tracking-widest bg-secondary px-2 py-0.5 rounded text-white flex items-center gap-1"><i data-lucide="user" class="w-3 h-3"></i> ${assignedName}</span>
                                ${t.deadline ? `<span class="text-[9px] uppercase tracking-widest ${isDone ? 'bg-secondary text-muted' : 'bg-red-500/20 text-red-400'} px-2 py-0.5 rounded flex items-center gap-1"><i data-lucide="calendar" class="w-3 h-3"></i> ${t.deadline}</span>` : ''}
                            </div>
                        </div>
                        <div class="shrink-0 flex items-center gap-1">${editBtn}${deleteBtn}</div>
                    </div>`;
                }).join('');
            } else {
                tasksHtml = `<p class="text-sm text-muted italic">Geen taken gevonden.</p>`;
            }
        } else {
            tasksHtml = `<p class="text-sm text-muted italic">Log in om je takenlijst te zien.</p>`;
        }

        container.innerHTML = `
            <div class="max-w-5xl mx-auto space-y-6 fade-in pb-10">
                <div class="w-full h-48 md:h-64 rounded-[2rem] overflow-hidden relative shadow-2xl mb-8 border border-white/5" style="${bannerStyle}">
                    <div class="absolute inset-0 bg-gradient-to-t from-black/90 via-black/40 to-transparent"></div>
                    <div class="absolute bottom-6 left-6 md:bottom-8 md:left-8 flex items-end gap-4 z-10">
                        ${logoHtml}
                        <div class="mb-1">
                            <p class="text-[10px] font-black uppercase text-papaya tracking-[0.3em] mb-1 drop-shadow-md">Command Center</p>
                            <h1 class="text-3xl md:text-5xl font-oswald uppercase tracking-wide text-white drop-shadow-lg">${league.name}</h1>
                        </div>
                    </div>
                </div>
                <div class="grid grid-cols-1 md:grid-cols-3 gap-4">
                    <div class="bg-card border border-border p-6 rounded-2xl text-center shadow-sm hover:border-papaya/30 transition-colors cursor-pointer" onclick="navigate('teams')"><p class="text-[10px] font-black uppercase text-muted tracking-widest mb-2">Actieve Teams</p><p class="text-4xl font-black">${teams.length}</p></div>
                    <div class="bg-card border border-border p-6 rounded-2xl text-center shadow-sm hover:border-papaya/30 transition-colors cursor-pointer" onclick="navigate('players')"><p class="text-[10px] font-black uppercase text-muted tracking-widest mb-2">Spelers in Systeem</p><p class="text-4xl font-black">${players.length}</p></div>
                    <div class="bg-card border border-border p-6 rounded-2xl text-center shadow-sm hover:border-papaya/30 transition-colors cursor-pointer" onclick="navigate('schedule')"><p class="text-[10px] font-black uppercase text-muted tracking-widest mb-2">Gespeelde Duels</p><p class="text-4xl font-black text-papaya">${schedule.filter(s => s.type === 'Game' && s.status === 'Final').length}</p></div>
                </div>
                <div class="grid grid-cols-1 lg:grid-cols-2 gap-6 mt-8">
                    <div class="bg-card border border-border p-6 md:p-8 rounded-3xl shadow-sm relative overflow-hidden">
                        <div class="absolute top-0 right-0 w-40 h-40 bg-papaya/5 rounded-full blur-3xl pointer-events-none"></div>
                        <h2 class="font-oswald text-xl uppercase tracking-wide border-b border-border pb-2 mb-6 flex items-center gap-2 text-white relative z-10"><i data-lucide="megaphone" class="w-5 h-5 text-papaya"></i> League Announcements</h2>
                        <div class="relative z-10">${announcementsHtml}</div>
                    </div>
                    <div class="bg-card border border-border p-6 md:p-8 rounded-3xl shadow-sm relative overflow-hidden">
                        <h2 class="font-oswald text-xl uppercase tracking-wide border-b border-border pb-2 mb-6 flex items-center gap-2 text-white relative z-10"><i data-lucide="check-square" class="w-5 h-5 text-papaya"></i> Actielijst</h2>
                        <div class="relative z-10">
                            ${tasksFormHtml}
                            <div class="space-y-3 custom-scrollbar max-h-[400px] overflow-y-auto pr-2">
                                ${tasksHtml}
                            </div>
                        </div>
                    </div>
                </div>
            </div>
        `;
    }
</script>
