    window.showCustomGameForm = function() {
        document.getElementById('sk-schedule-list').classList.add('hidden');
        document.getElementById('sk-custom-form').classList.remove('hidden');
    };

    window.hideCustomGameForm = function() {
        document.getElementById('sk-schedule-list').classList.remove('hidden');
        document.getElementById('sk-custom-form').classList.add('hidden');
    };

    window.submitCustomGame = function() {
        const homeTeamId = document.getElementById('cg-home-team').value;
        const awayTeamId = document.getElementById('cg-away-team').value;

        if(!homeTeamId || !awayTeamId) {
            alert('Please select both a Home Team and an Away Team.');
            return;
        }
        if(homeTeamId === awayTeamId) {
            alert('Home and Away teams cannot be the same.');
            return;
        }

        const customConfig = {
            isCustom: true,
            homeTeamId,
            awayTeamId,
            season: document.getElementById('cg-season').value,
            leagueId: document.getElementById('cg-league').value,
            division: document.getElementById('cg-division').value,
            ruleset: document.getElementById('cg-ruleset').value,
            scheduleId: document.getElementById('cg-schedule-id').value,
            periodLength1: document.getElementById('cg-period-length-1').value,
            periodLength2: document.getElementById('cg-period-length-2').value,
            periodLength3: document.getElementById('cg-period-length-3').value,
            periods: document.getElementById('cg-periods').value,
            enableOt: document.getElementById('cg-enable-ot').value,
            otPeriods: document.getElementById('cg-ot-periods').value,
            otPeriodLength: document.getElementById('cg-ot-period-length').value,
            enableShootout: document.getElementById('cg-enable-shootout').value,
            shootoutAttempts: document.getElementById('cg-shootout-attempts').value,
            rotateShooters: document.getElementById('cg-rotate-shooters').value,
            gameDate: document.getElementById('cg-game-date').value,
            gameStartTime: document.getElementById('cg-game-start-time').value,
            countedAttendance: document.getElementById('cg-counted-attendance').value,
            officialAttendance: document.getElementById('cg-official-attendance').value,
            officials: document.getElementById('cg-officials').value,
            scorekeeper: document.getElementById('cg-scorekeeper').value,
            rosters: document.getElementById('cg-rosters').value,
            lines: document.getElementById('cg-lines').value,
            salaryCap: document.getElementById('cg-salary-cap').value,
            autobalance: document.getElementById('cg-autobalance').value,
            gameMvp: document.getElementById('cg-game-mvp').value,
            timing: document.getElementById('cg-timing').value,
            penaltyRegistration: document.getElementById('cg-penalty-registration').value,
            penaltyClock: document.getElementById('cg-penalty-clock').value
        };

        startScorekeeperConsole('custom', customConfig);
    };

    function initScorekeeperUI(gameId, customConfig) {
        let homeTeamId = 'LII';
        let awayTeamId = 'TOR';

        if (gameId === 'custom' && customConfig) {
            homeTeamId = customConfig.homeTeamId;
            awayTeamId = customConfig.awayTeamId;
        } else if (gameId) {
            const scheduledGame = MockData.schedule.find(s => s.id === gameId);
            if (scheduledGame) {
                homeTeamId = scheduledGame.homeTeamId;
                awayTeamId = scheduledGame.awayTeamId;
            }
        }

        const homeTeam = getTeam(homeTeamId) || { id: homeTeamId, name: homeTeamId, shortName: homeTeamId.substring(0,3).toUpperCase() };
        const awayTeam = getTeam(awayTeamId) || { id: awayTeamId, name: awayTeamId, shortName: awayTeamId.substring(0,3).toUpperCase() };

        const htId = homeTeam.id;
        const atId = awayTeam.id;
        const htShort = homeTeam.shortName || homeTeam.name.substring(0,3).toUpperCase();
        const atShort = awayTeam.shortName || awayTeam.name.substring(0,3).toUpperCase();
        const htInitial = htShort.charAt(0);
        const atInitial = atShort.charAt(0);

// --- State Management ---
            const AppState = {
                clockSeconds: 118,
                isRunning: false,
                clockInterval: null,
                stats: {},
                ui: {
                    selectedPenaltyTeam: htId,
                    selectedGoalTeam: htId
                }
            };
            AppState.stats[htId] = { shots: 15, score: 2 };
            AppState.stats[atId] = { shots: 12, score: 4 };


            // --- DOM Elements ---
            const DOM = {
                clock: {
                    main: document.getElementById('match-clock'),
                    modal: document.getElementById('modal-clock-display'),
                    btnPlus: document.getElementById('btn-plus-1s'),
                    btnMinus: document.getElementById('btn-minus-1s')
                },
                scoreboard: {
                    homeScore: document.getElementById('score-home'),
                    awayScore: document.getElementById('score-away'),
                    homeShots: document.getElementById('shots-home'),
                    awayShots: document.getElementById('shots-away')
                },
                rink: {
                    container: document.getElementById('interactive-rink'),
                    areaLii: document.getElementById('rink-lii'),
                    areaTor: document.getElementById('rink-tor')
                },
                navigation: {
                    controls: document.querySelectorAll('.control-btn'),
                    tabs: document.querySelectorAll('.tab-btn'),
                    viewsContainer: document.getElementById('views-container'),
                    mainViews: [
                        document.getElementById('view-pre-game'),
                        document.getElementById('view-match'),
                        document.getElementById('view-post-game')
                    ],
                    headerTabs: document.getElementById('header-tabs')
                },
                modal: {
                    container: document.getElementById('whistle-modal'),
                    btnWhistle: document.getElementById('btn-whistle'),
                    btnClose: document.getElementById('close-modal'),
                    btnResume: document.getElementById('resume-play'),
                    actionBtns: document.querySelectorAll('.modal-action-btn')
                },
                feed: {
                    list: document.getElementById('play-by-play-list')
                }
            };

            // --- GAS Integration Hooks ---
            const GASHooks = {
                logEvent: (eventType, details) => {
                    console.log(`[GAS Hook] Event: ${eventType}`, details);
                    if (typeof google !== 'undefined' && google.script) {
                        google.script.run
                            .withSuccessHandler((res) => console.log('GAS Success:', res))
                            .withFailureHandler((err) => console.error('GAS Error:', err))
                            .processHockeyEvent({ type: eventType, data: details, timestamp: new Date() });
                    }
                }
            };

            // --- UI Updates & Logic ---
            function updateClockDisplay() {
                const mins = Math.floor(AppState.clockSeconds / 60);
                const secs = AppState.clockSeconds % 60;
                const displayStr = `${mins.toString().padStart(2, '0')}:${secs.toString().padStart(2, '0')}`;
                if (DOM.clock.main) DOM.clock.main.textContent = displayStr;
                if (DOM.clock.modal) DOM.clock.modal.textContent = displayStr;
            }

            function startClock() {
                if (!AppState.isRunning) {
                    AppState.isRunning = true;
                    GASHooks.logEvent('CLOCK_START', { timeRemaining: AppState.clockSeconds });
                    AppState.clockInterval = setInterval(() => {
                        if (AppState.clockSeconds > 0) {
                            AppState.clockSeconds--;
                            updateClockDisplay();
                        } else {
                            stopClock();
                        }
                    }, 1000);
                }
            }

            function stopClock() {
                AppState.isRunning = false;
                if (AppState.clockInterval) clearInterval(AppState.clockInterval);
                GASHooks.logEvent('CLOCK_STOP', { timeRemaining: AppState.clockSeconds });
            }

            // --- Event Listeners ---

            // Rink Interaction
            DOM.rink.container.addEventListener('click', (e) => {
                const rect = DOM.rink.container.getBoundingClientRect();
                const x = e.clientX - rect.left;
                const y = e.clientY - rect.top;

                const pctX = (x / rect.width) * 100;
                const team = pctX < 50 ? htId : atId;
                const color = team === htId ? '#e67e22' : '#004b8d';

                AppState.stats[team].shots++;
                if (team === htId) DOM.scoreboard.homeShots.textContent = `${AppState.stats[htId].shots} shots`;
                else DOM.scoreboard.awayShots.textContent = `${AppState.stats[atId].shots} shots`;

                GASHooks.logEvent('SHOT_RECORDED', { team, x, y });

                const marker = document.createElement('div');
                marker.className = 'absolute rounded-full rink-marker z-50';
                marker.style.width = '12px';
                marker.style.height = '12px';
                marker.style.backgroundColor = color;
                marker.style.left = `${x - 6}px`;
                marker.style.top = `${y - 6}px`;
                marker.style.boxShadow = `0 0 10px ${color}`;

                DOM.rink.container.appendChild(marker);
                setTimeout(() => {
                    if (DOM.rink.container.contains(marker)) DOM.rink.container.removeChild(marker);
                }, 1000);
            });

            // Navigation Flow
            DOM.navigation.controls.forEach(btn => {
                btn.addEventListener('click', () => {
                    DOM.navigation.controls.forEach(b => b.classList.remove('active'));
                    btn.classList.add('active');

                    DOM.navigation.mainViews.forEach(v => { if(v) v.classList.add('hidden'); });

                    const targetId = btn.getAttribute('data-target');
                    const targetView = document.getElementById(targetId);
                    if(targetView) targetView.classList.remove('hidden');

                    if (targetId === 'view-match') {
                        DOM.navigation.headerTabs.classList.remove('hidden');
                        DOM.navigation.headerTabs.classList.add('flex');
                    } else {
                        DOM.navigation.headerTabs.classList.add('hidden');
                        DOM.navigation.headerTabs.classList.remove('flex');
                    }

                    if (btn.textContent.trim() === 'Match') {
                        startClock();
                    } else {
                        stopClock();
                    }
                });
            });

            DOM.navigation.tabs.forEach(btn => {
                btn.addEventListener('click', () => {
                    DOM.navigation.tabs.forEach(b => {
                        b.classList.remove('text-primary', 'border-b-2', 'border-primary', 'active-tab');
                        b.classList.add('text-on-surface-variant');
                    });

                    btn.classList.add('text-primary', 'border-b-2', 'border-primary', 'active-tab');
                    btn.classList.remove('text-on-surface-variant');

                    const targetId = btn.getAttribute('data-target');
                    DOM.navigation.viewsContainer.querySelectorAll(':scope > div').forEach(view => {
                        view.classList.add('hidden');
                    });
                    document.getElementById(targetId).classList.remove('hidden');
                });
            });

            // Modal Controls
            function openModal() {
                stopClock();
                updateClockDisplay();
                DOM.modal.container.classList.remove('hidden');
            }

            function closeModal() {
                DOM.modal.container.classList.add('hidden');
            }

            DOM.modal.btnWhistle.addEventListener('click', openModal);
            DOM.modal.btnClose.addEventListener('click', closeModal);

            DOM.modal.btnResume.addEventListener('click', () => {
                closeModal();
                if(document.querySelector('.control-btn.active').textContent.trim() === 'Match') {
                    startClock();
                }
            });

            DOM.clock.btnPlus.addEventListener('click', () => {
                AppState.clockSeconds++;
                updateClockDisplay();
                GASHooks.logEvent('CLOCK_ADJUSTED', { adjustment: '+1s', newTime: AppState.clockSeconds });
            });
            DOM.clock.btnMinus.addEventListener('click', () => {
                if (AppState.clockSeconds > 0) {
                    AppState.clockSeconds--;
                    updateClockDisplay();
                    GASHooks.logEvent('CLOCK_ADJUSTED', { adjustment: '-1s', newTime: AppState.clockSeconds });
                }
            });

            DOM.modal.actionBtns.forEach(btn => {
                btn.addEventListener('click', () => {
                    const targetId = btn.getAttribute('data-target');
                    const targetDiv = document.getElementById(targetId);
                    const icon = btn.querySelector('span');

                    if (targetDiv.classList.contains('hidden')) {
                        document.querySelectorAll('[id$="-options"]').forEach(div => div.classList.add('hidden'));
                        document.querySelectorAll('.modal-action-btn span').forEach(sp => sp.textContent = 'expand_more');
                        targetDiv.classList.remove('hidden');
                        icon.textContent = 'expand_less';
                    } else {
                        targetDiv.classList.add('hidden');
                        icon.textContent = 'expand_more';
                    }
                });
            });

            // Feed Updates
            function addEventToFeed(team, title, subtitle, badgeText = '') {
                const timeStr = DOM.clock.main.textContent;
                const isHome = team === htId;
                const teamInitial = isHome ? htInitial : atInitial;
                const bgClass = isHome ? 'bg-team-lii' : 'bg-team-tor';
                const textClass = isHome ? 'text-team-lii' : 'text-team-tor';

                const scoreStr = `${htShort} ${AppState.stats[htId].score} - ${atShort} ${AppState.stats[atId].score}`;

                const eventCard = document.createElement('div');
                eventCard.className = 'flex items-center gap-4 bg-surface-container-high p-3 rounded-lg border border-outline-variant';

                let badgeHtml = '';
                if (badgeText) {
                    badgeHtml = `<div class="text-label-caps font-label-caps ${textClass} font-bold">${badgeText}</div>`;
                }

                eventCard.innerHTML = `
                    <div class="w-12 h-12 rounded-full ${bgClass} flex items-center justify-center shadow-sm">
                        <span class="text-headline-lg font-headline-lg text-white font-bold">${teamInitial}</span>
                    </div>
                    <div class="flex-1 flex flex-col">
                        <span class="text-body-lg font-body-lg font-bold text-on-surface">${title}</span>
                        <span class="text-body-sm font-body-sm text-on-surface-variant">${subtitle}</span>
                        <span class="text-label-caps font-label-caps text-outline mt-1">${timeStr} ${scoreStr}</span>
                    </div>
                    ${badgeHtml}
                `;

                const header = DOM.feed.list.querySelector('h3');
                if (header) header.insertAdjacentElement('afterend', eventCard);
                else DOM.feed.list.appendChild(eventCard);
            }

            // Action Handlers
            document.querySelectorAll('.faceoff-log-btn').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    const team = e.target.getAttribute('data-team');
                    addEventToFeed(team, `Face-off Won`, `${team} secures the puck`);
                    GASHooks.logEvent('FACEOFF_WON', { team });
                    closeModal();
                    if(document.querySelector('.control-btn.active').textContent.trim() === 'Match') startClock();
                });
            });

            document.querySelectorAll('.penalty-team-btn').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    document.querySelectorAll('.penalty-team-btn').forEach(b => b.classList.remove('bg-surface-variant', 'border-primary', 'font-bold'));
                    e.target.classList.add('bg-surface-variant', 'border-primary', 'font-bold');
                    AppState.ui.selectedPenaltyTeam = e.target.getAttribute('data-team');
                });
            });

            document.getElementById('log-penalty-btn')?.addEventListener('click', () => {
                const selects = document.querySelectorAll('.penalty-log-select');
                const player = selects[0].value || 'Unknown Player';
                const type = selects[1].value || 'Unknown Penalty';
                addEventToFeed(AppState.ui.selectedPenaltyTeam, `${player}`, `${type} (2 min)`, 'PEN');
                GASHooks.logEvent('PENALTY_LOGGED', { team: AppState.ui.selectedPenaltyTeam, player, type });
                closeModal();
                if(document.querySelector('.control-btn.active').textContent.trim() === 'Match') startClock();
            });

            document.querySelectorAll('.goal-team-btn').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    document.querySelectorAll('.goal-team-btn').forEach(b => b.classList.remove('bg-surface-variant', 'border-primary', 'font-bold'));
                    e.target.classList.add('bg-surface-variant', 'border-primary', 'font-bold');
                    AppState.ui.selectedGoalTeam = e.target.getAttribute('data-team');
                });
            });

            document.getElementById('log-goal-btn')?.addEventListener('click', () => {
                const selects = document.querySelectorAll('.goal-log-select');
                const scorer = selects[0].value || 'Unknown Scorer';
                const ast1 = selects[1].value;
                const ast2 = selects[2].value;

                let assistText = 'Unassisted';
                if (ast1 && ast1 !== 'None' && ast2 && ast2 !== 'None') assistText = `Ast: ${ast1}, ${ast2}`;
                else if (ast1 && ast1 !== 'None') assistText = `Ast: ${ast1}`;
                else if (ast2 && ast2 !== 'None') assistText = `Ast: ${ast2}`;

                const team = AppState.ui.selectedGoalTeam;
                AppState.stats[team].score++;

                if (team === htId) DOM.scoreboard.homeScore.textContent = AppState.stats[htId].score;
                else DOM.scoreboard.awayScore.textContent = AppState.stats[atId].score;

                addEventToFeed(team, `${scorer}`, assistText, 'GOAL');
                GASHooks.logEvent('GOAL_LOGGED', { team, scorer, assistText });

                closeModal();
                if(document.querySelector('.control-btn.active').textContent.trim() === 'Match') startClock();
            });

            // Initialization
            updateClockDisplay();
    }
</script>
