    ],

      ['r1', 'G-003', 'p1', 'Attending'],
      ['r2', 'G-003', 'p3', 'Declined']

      ['n1', 'u1', 'Vergeet de pucks niet', '2024-10-10']
    ]
  };

  // 4. Vul de sheets met tabbladen, headers en testdata
  Object.keys(MAP).forEach(category => {
    const doc = docs[category];
    const tabs = MAP[category].tabs;

    Object.keys(tabs).forEach(tabKey => {
      const tabInfo = tabs[tabKey];
      let sheet = doc.insertSheet(tabInfo.name);

      // Maak de headers op basis van de kolomnummers
      let headers = [];
      Object.keys(tabInfo.cols).forEach(colName => {
        headers[tabInfo.cols[colName] - 1] = colName;
      });

      // Maak slimme testdata aan
      let testData = headers.map(h => {
        if (!h) return '';
        if (h === 'leagueId') return 'GLOBAL';
        if (h === 'settingKey') return 'primaryColor';
        if (h === 'settingValue') return '#facc15'; // Gele testkleur
        if (h === 'email') return 'test@manager.nl';
        if (h === 'role') return 'Admin';
        if (h.toLowerCase().includes('id')) return '1';
        if (h.toLowerCase().includes('score')) return '3';
        return 'Test_' + h;
      });

      sheet.appendRow(headers);

      // Vul de rijen als er seedData is, anders één generieke fake row
      if (seedData[tabInfo.name]) {
        seedData[tabInfo.name].forEach(row => {
          // Zorg dat de array net zo lang is als de headers
          let paddedRow = headers.map((h, i) => row[i] !== undefined ? row[i] : '');
          sheet.appendRow(paddedRow);
        });
      } else {
        sheet.appendRow(testData);
      }

      // Maak de header vetgedrukt
      sheet.getRange(1, 1, 1, headers.length).setFontWeight("bold");
    });

    // Verwijder het standaard lege 'Blad1' of 'Sheet1'
    const defaultSheet = doc.getSheetByName("Sheet1") || doc.getSheetByName("Blad1");
    if (defaultSheet) doc.deleteSheet(defaultSheet);
  });

  Logger.log("Succes! Sheets zijn aangemaakt en de ID's staan in je Script Properties.");
}

function addMissingTabsToExisting() {
  const props = PropertiesService.getScriptProperties();

  // Haal het ID op van je bestaande DATA sheet
  const dataSheetId = props.getProperty('SHEET2_ID');

  if (!dataSheetId) {
    Logger.log("Fout: Geen bestaande DATA sheet gevonden in de instellingen.");
    return;
  }

  const doc = SpreadsheetApp.openById(dataSheetId);

  // De tabbladen en headers die we misten
  const missingTabs = {
    'Announcements': ['id', 'leagueId', 'title', 'content', 'date', 'authorId'],
    'Tasks': ['id', 'leagueId', 'title', 'description', 'assignedTo', 'deadline', 'status'],
    'RSVPs': ['id', 'eventId', 'playerId', 'status'],
    'GameRoles': ['id', 'gameId', 'userId', 'role', 'status'],
    'Notes': ['id', 'authorId', 'content', 'date']
  };

  // Loop door de missende tabbladen heen
  Object.keys(missingTabs).forEach(tabName => {
    // Check of het tabblad al bestaat om dubbelingen te voorkomen
    if (!doc.getSheetByName(tabName)) {
      let sheet = doc.insertSheet(tabName);

      // Voeg de headers toe
      sheet.appendRow(missingTabs[tabName]);

      // Maak de headers vetgedrukt
      sheet.getRange(1, 1, 1, missingTabs[tabName].length).setFontWeight("bold");

      Logger.log("✅ Succesvol toegevoegd: " + tabName);
    } else {
      Logger.log("⚠️ Bestond al, overgeslagen: " + tabName);
    }
  });
}

function forceFixAdminUser() {
  const id = PropertiesService.getScriptProperties().getProperty('SHEET1_ID');
  const sheet = SpreadsheetApp.openById(id).getSheetByName('Users');

  // 1. Forceer de juiste headers op Rij 1
  sheet.getRange(1, 1, 1, 8).setValues([
    ['userId', 'email', 'role', 'linkedLeagueId', 'status', 'username', 'name', 'pincode']
  ]);

  // 2. Forceer de admin data op Rij 2
  sheet.getRange(2, 1, 1, 8).setValues([
    ['u1', 'admin@league.nl', 'Officer', 'l1', 'Approved', 'admin', 'League Admin', '1234']
  ]);

  Logger.log("✅ Actieve database geforceerd bijgewerkt met de juiste admin data!");
}
